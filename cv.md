function alphabetPosition(text) {
  const alfabet = "abcdefghijklmnopqrstuvwxyz"
  let result = "";
  for (let i = 0; i < text.length; i += 1) {
    text = text.toLowerCase();
    if (alfabet.indexOf(text[i]) !== -1) {
      result += alfabet.indexOf(text[i]) + 1;
      result += " ";
    }
  }
  return result.trim();
}
